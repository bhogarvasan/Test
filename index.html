class BybitWebSocketClient {

  constructor(network = 'mainnet') {

    this.network = network;

    this.endpoint = network === 'mainnet' 

      ? 'wss://stream.bybit.com/v5/public/linear' 

      : 'wss://stream-testnet.bybit.com/v5/public/linear';

    this.privateEndpoint = network === 'mainnet'

      ? 'wss://stream.bybit.com/v5/private'

      : 'wss://stream-testnet.bybit.com/v5/private';

  }


  connect() {

    if (G.ws && (G.ws.readyState === WebSocket.OPEN || G.ws.readyState === WebSocket.CONNECTING)) {

      return;

    }

    

    G.wsConnecting = true;

    log('i', `🔌 Initializing low-latency WebSocket connection (${this.network})...`);

    

    try {

      G.ws = new WebSocket(this.endpoint);

      

      G.ws.onopen = () => {

        G.wsConnecting = false;

        G.wsRetryCount = 0;

        log('s', '🔌 WebSocket Connection established successfully.');

        this.startHeartbeat();

        this.resubscribeAll();

      };

      

      G.ws.onmessage = (event) => {

        this.handleMessage(JSON.parse(event.data));

      };

      

      G.ws.onerror = (error) => {

        log('e', `🔌 WebSocket Error: ${error.message || 'Unknown network error'}`);

      };

      

      G.ws.onclose = () => {

        G.wsConnecting = false;

        this.stopHeartbeat();

        log('w', '🔌 WebSocket Connection closed. Attempting auto-reconnection...');

        this.reconnect();

      };

    } catch(e) {

      G.wsConnecting = false;

      log('e', `🔌 Failed to build WebSocket client: ${e.message}`);

      this.reconnect();

    }

  }


  reconnect() {

    if (G.wsConnecting) return;

    const delay = Math.min(30000, Math.pow(2, G.wsRetryCount) * 1000);

    G.wsRetryCount++;

    log('i', `🔌 Reconnecting in ${(delay / 1000).toFixed(0)}s (Attempt #${G.wsRetryCount})`);

    setTimeout(() => this.connect(), delay);

  }


  startHeartbeat() {

    G.wsPingInterval = setInterval(() => {

      if (G.ws && G.ws.readyState === WebSocket.OPEN) {

        G.ws.send(JSON.stringify({ op: "ping" }));

      }

    }, 20000);

  }


  stopHeartbeat() {

    if (G.wsPingInterval) {

      clearInterval(G.wsPingInterval);

      G.wsPingInterval = null;

    }

  }


  subscribe(topic) {

    if (G.activeSubscriptions.has(topic)) return;

    G.activeSubscriptions.add(topic);

    this.sendPayload({ op: "subscribe", args: [topic] });

  }


  unsubscribe(topic) {

    if (!G.activeSubscriptions.has(topic)) return;

    G.activeSubscriptions.delete(topic);

    this.sendPayload({ op: "unsubscribe", args: [topic] });

  }


  resubscribeAll() {

    if (G.activeSubscriptions.size === 0) return;

    const args = Array.from(G.activeSubscriptions);

    this.sendPayload({ op: "subscribe", args });

    log('i', `🔌 Restored ${args.length} real-time data subscriptions.`);

  }


  sendPayload(payload) {

    if (G.ws && G.ws.readyState === WebSocket.OPEN) {

      G.ws.send(JSON.stringify(payload));

    }

  }


  handleMessage(msg) {

    if (msg.op === 'pong') return;

    if (!msg.topic || !msg.data) return;


    const parts = msg.topic.split('.');

    const streamType = parts[0]; // e.g. "kline" or "tickers"

    const param = parts[1];      // resolution (e.g. "1") or symbol

    const symbol = parts[parts.length - 1];


    if (streamType === 'kline') {

      const interval = param;

      const candles = Array.isArray(msg.data) ? msg.data : [msg.data];

      this.updateKlineCache(symbol, interval, candles);

    } else if (streamType === 'tickers') {

      this.updateTickerCache(symbol, msg.data);

    }

  }


  updateKlineCache(symbol, interval, rawCandles) {

    const cacheMap = interval === '1' ? G.kline1mCache : G.kline15mCache;

    if (!cacheMap[symbol]) {

      cacheMap[symbol] = [];

    }


    const currentArray = cacheMap[symbol];

    

    for (const raw of rawCandles) {

      // Bybit format: [start, open, high, low, close, volume, turnover]

      const mapped = [

        raw.start.toString(),

        raw.open,

        raw.high,

        raw.low,

        raw.close,

        raw.volume,

        raw.turnover

      ];


      const index = currentArray.findIndex(c => c[0] === mapped[0]);

      if (index !== -1) {

        // Update forming candle

        currentArray[index] = mapped;

      } else {

        // Append new candle and keep within boundary lengths

        currentArray.push(mapped);

        const maxLen = interval === '1' ? 65 : 250;

        if (currentArray.length > maxLen) {

          currentArray.shift();

        }

      }

    }

  }


  updateTickerCache(symbol, data) {

    if (!G.tickerCache[symbol]) G.tickerCache[symbol] = {};

    Object.assign(G.tickerCache[symbol], data);

  }

          const res = JSON.parse(event.data);

          if (res.reqId === reqId) {

            G.wsOrderChannel.removeEventListener('message', responseHandler);

            if (res.retCode === 0) {

              resolve(res);

            } else {

              reject(new Error(`[Bybit WS Reject] ${res.retMsg}`));

            }

          }

        };


        G.wsOrderChannel.addEventListener('message', responseHandler);

        G.wsOrderChannel.send(JSON.stringify(payload));


        // 5s hard timeout on WS order confirmation

        setTimeout(() => {

          G.wsOrderChannel.removeEventListener('message', responseHandler);

          reject(new Error('WebSocket Order Confirmation Timeout'));

        }, 5000);


      } catch (err) {

        reject(err);

      }

    });

  }


  connectPrivate() {

    if (G.wsOrderChannel && (G.wsOrderChannel.readyState === WebSocket.OPEN || G.wsOrderChannel.readyState === WebSocket.CONNECTING)) {

      return;

    }

    log('i', '🔐 Activating private WebSocket order routing...');

    try {

      G.wsOrderChannel = new WebSocket(this.privateEndpoint);

      G.wsOrderChannel.onopen = async () => {

        // Authenticate Private Connection

        const ts = Date.now().toString();

        const signature = await sign(S.apiSecret, "GET/auth" + ts + "5000");

        G.wsOrderChannel.send(JSON.stringify({

          op: "auth",

          args: [S.apiKey, "5000", signature]

        }));

        log('s', '🔐 Private Order Channel Authenticated.');

      };

      G.wsOrderChannel.onclose = () => {

        log('w', '🔐 Private order stream disconnected. Re-initializing...');

        setTimeout(() => this.connectPrivate(), 5000);

      };

    } catch(e) {

    }

  }

// Re-map REST calls during live scans to pull directly from the low-latency WS Cache

async function getCachedKline(symbol, interval, limit = 60) {

  const cacheMap = interval === '1' ? G.kline1mCache : G.kline15mCache;

  

  if (cacheMap[symbol] && cacheMap[symbol].length >= limit) {

    return { result: { list: cacheMap[symbol].slice(-limit).reverse() } };

  }

  

  // Cache miss or insufficient depth: fetch via REST and prime memory

  const path = `/v5/market/kline?category=linear&symbol=${symbol}&interval=${interval}&limit=${Math.max(limit, 250)}`;

  const restData = await pubCached(path);

  if (restData?.result?.list) {

    const listCopy = [...restData.result.list].reverse();

    cacheMap[symbol] = listCopy;

  }

  return restData;

}

async function syncWebSocketSubscriptions(focusList) {

  for (const topic of G.activeSubscriptions) {

    }

  }

  for (const pair of focusList) {

    const kline1Topic = `kline.1.${sym}`;

    const tickerTopic = `tickers.${sym}`;

    }

  }

}

// Top 20% volume ranking filter (Relative Volume)

function checkRelativeVolume(vols, curVol) {

  if (vols.length < 20) return false;

  const sorted = [...vols].sort((a, b) => a - b);

  const index80Pct = Math.floor(sorted.length * 0.80);

  const thresholdVal = sorted[index80Pct];

  return curVol >= thresholdVal;

}


// 1m Fractal Pivot Confirmation & Market Structure Shift (MSS)

// Long: Candle[-1].close > Candle[-2].high, where Candle[-2].high is a local fractal peak

// Short: Candle[-1].close < Candle[-2].low, where Candle[-2].low is a local fractal valley

function detect1MMSS(k1c, direction) {

  if (k1c.length < 6) return { confirmed: false, level: 0 };

  

  // Index alignments on reversed array (list[0] = oldest, list[len-1] = current forming)

  const lastIndex = k1c.length - 1;

  const c1 = k1c[lastIndex - 1]; // Candle[-1] (completed)

  const c2 = k1c[lastIndex - 2]; // Candle[-2]

  const c3 = k1c[lastIndex - 3]; // Candle[-3]


  const c1Close = parseFloat(c1[4]);

  const c2High  = parseFloat(c2[2]);

  const c2Low   = parseFloat(c2[3]);

  const c1High  = parseFloat(c1[2]);

  const c3High  = parseFloat(c3[2]);

  const c1Low   = parseFloat(c1[3]);

  const c3Low   = parseFloat(c3[3]);


  if (direction === 'LONG') {

    // Validate Candle[-2] is a fractal high

    const isFractalHigh = (c2High > c1High) && (c2High > c3High);

    // MSS: Candle[-1].close closes above the Fractal High

    const breakConfirmed = c1Close > c2High;

    return { confirmed: isFractalHigh && breakConfirmed, level: c2High };

  } else {

    // Validate Candle[-2] is a fractal low

    const isFractalLow = (c2Low < c1Low) && (c2Low < c3Low);

    // MSS: Candle[-1].close closes below the Fractal Low

    const breakConfirmed = c1Close < c2Low;

    return { confirmed: isFractalLow && breakConfirmed, level: c2Low };

  }

}

    // FILTER 1 — 15m Trend Bias Gate (EMA 200)

    // ─────────────────────────────────────────────────────────────

    const ema200on15 = calcEMA(closes15, 200);

    let direction = 'NEUTRAL';

    if (price > ema200on15) {

      direction = 'LONG';

    } else if (price < ema200on15) {

      direction = 'SHORT';

    } else {

      return null; // Trend is completely flat / unresolvable

    }


    // ─────────────────────────────────────────────────────────────

    // FILTER 2 — 1m Liquidity Sweep (lookback=5)

    // ─────────────────────────────────────────────────────────────

    // Compare last closed candle (k1c[len-2]) against previous 5 candles (k1c[len-7] to k1c[len-3])

    const lastClosedCandle = k1c[k1c.length - 2];

    const lcLow = parseFloat(lastClosedCandle[3]);

    const lcHigh = parseFloat(lastClosedCandle[2]);

    const referenceWindow = k1c.slice(-7, -2);

    

    let swept = false;

    if (direction === 'LONG') {

      const minLow = Math.min(...referenceWindow.map(k => parseFloat(k[3])));

      if (lcLow < minLow) swept = true;

    } else {

      const maxHigh = Math.max(...referenceWindow.map(k => parseFloat(k[2])));

      if (lcHigh > maxHigh) swept = true;

    }

    if (!swept) return null; // No sweep detected = fail waterfall


    // ─────────────────────────────────────────────────────────────

    // FILTER 3 — 1m Market Structure Shift (MSS Pivot Confirmation)

    // ─────────────────────────────────────────────────────────────

    const mssCheck = detect1MMSS(k1c, direction);

          if (tpRes.retCode === 0) {

            t.tpOrderId = tpRes.result?.orderId || null;

            log('i', `🎯 Rested limit TP placed for ${result.symbol} @ ${fp(tp1Price)}`);

          }

        } catch(e) {}


        saveState();

        updateTradeLog(); updateStats(); updateLivePanel();

        log('s', `✅ EXECUTION FILLED: ${result.symbol} ${result.direction} @ ${actualEntry.toFixed(6)} | SL: ${finalSlPrice.toFixed(6)}`);

        playAudio('entry');


      } finally {

        _pendingExecutions.delete(result.symbol);

        updateStats();

      }

    } finally {

      unlock();

    }

  } catch(e) {

    log('e', `${result.symbol} Execution Error: ` + e.message);

  } finally {

    _inFlightOrders.delete(result.symbol);

  }

}

async function monitorPositions() {

  if (!S.running) return;

  try {

    const live = await getPositions();


    // Reconcile and process active trades

    const liveSyms = live.map(p => p.symbol);

    for (let i = S.openPositions.length - 1; i >= 0; i--) {

      const pos = S.openPositions[i];

      const lp = live.find(p => p.symbol === pos.symbol);

      

      if (!lp) {

        // Position closed externally

        try { await api('POST', '/v5/order/cancel-all', { category: 'linear', symbol: pos.symbol }); } catch(e) {}

        const pnl = await getClosedPnl(pos.symbol, pos.openTime);

        pos.pnl = pnl;

        pos.status = pnl >= 0 ? 'closed' : 'stopped';

        S.sessionPnl += pnl;

        S.totalPnl += pnl;

        if (pnl >= 0) { S.wins++; } else { S.losses++; }

        S.openPositions.splice(i, 1);

        saveState();

        log(pnl >= 0 ? 's' : 'w', `${pnl >= 0 ? '✅' : '❌'} ${pos.symbol} closed. P&L: ${pnl.toFixed(4)} USDT`);

        if (pnl < 0) playAudio('sl'); else playAudio('tp');

        updateTradeLog(); updateStats(); updateLivePanel();

        continue;

      }


      // Live metrics update

      const curPrice = parseFloat(lp.markPrice || lp.avgPrice || pos.entryPrice);

      pos.livePnl = parseFloat(lp.unrealisedPnl || 0);

      pos.currentPrice = curPrice;


      // ─────────────────────────────────────────────────────────────

      // VOLATILITY-CLIMAX BREAK-EVEN TRAIL TRIGGER

      // If profit target is 50% reached AND 1m volume > 2x average volume,

      // move Stop Loss to Break-Even (entryPrice)

      // ─────────────────────────────────────────────────────────────

      if (!pos.breakeven) {

        const tpDistance = Math.abs(pos.tp - pos.entryPrice);

        const currentProfit = Math.abs(curPrice - pos.entryPrice);

        const progressPct = currentProfit / tpDistance;


        if (progressPct >= 0.50) {

          // Verify 1m Volume condition (> 2x average volume of last 20 candles)

          const k1Data = G.kline1mCache[pos.symbol];

          if (k1Data && k1Data.length >= 21) {

            const vols = k1Data.slice(-21, -1).map(k => parseFloat(k[5]));

            const avgVol20 = vols.reduce((a, b) => a + b, 0) / 20;

            const currentVol = parseFloat(k1Data[k1Data.length - 1][5]);


            if (currentVol > avgVol20 * 2.0) {

              // Volatility-Climax detected! Move Stop Loss to Break-even

              try {

                const step = pos.pair?.tickSize || 0.0001;

                const dec = stepDecimals(step);

                const bePrice = pos.entryPrice.toFixed(dec);


                const res = await api('POST', '/v5/order/amend', {

                  category: 'linear',

                  symbol: pos.symbol,

                  orderId: pos.id, // Amends the active stop loss order associated with position

                  stopLoss: bePrice

                });


                if (res.retCode === 0 || res.retCode === 110043) {

                  pos.breakeven = true;

                  pos.sl = pos.entryPrice;

                  log('s', `🔥 [VOL-CLIMAX] ${pos.symbol} - Profit target 50% hit with volume spike! SL moved to Breakeven (${bePrice})`);

                  saveState();

                }

              } catch (e) {

                log('e', `[VOL-CLIMAX ERROR] Failed to amend SL for ${pos.symbol}: ${e.message}`);

              }

            }

          }

        }

      }

    }

    updateLivePanel();

  } catch (e) {

    log('e', 'monitorPositions error: ' + e.message);

  }

}
